159 ARCCORE_HOST_DEVICE inline Real2
160 computeNormalFace(FaceLocalId face_lid,
161  IndexedFaceNodeConnectivityView fn_cv,
162  Accelerator::VariableNodeReal3InView in_node_coord,
163  FaceInfoListView faces_infos)
164 {
165  Real3 n0 = in_node_coord[fn_cv.nodeId(face_lid, 0)];
166  Real3 n1 = in_node_coord[fn_cv.nodeId(face_lid, 1)];
167 
168  if (!faces_infos.isSubDomainBoundaryOutside(face_lid)) {
169  Real3 tmp = n0;
170  n0 = n1;
171  n1 = tmp;
172  }
173 
174  Real2 N;
175  Real norm_N = math::sqrt((n1.y - n0.y) * (n1.y - n0.y) + (n1.x - n0.x) * (n1.x - n0.x));
176  N.x = (n1.y - n0.y) / norm_N;
177  N.y = (n0.x - n1.x) / norm_N;
178  return N;
179 }

190 ARCCORE_HOST_DEVICE inline Real3
191 computeNormalTriangle(FaceLocalId face_lid,
192  IndexedFaceNodeConnectivityView fn_cv,
193  Accelerator::VariableNodeReal3InView in_node_coord,
194  FaceInfoListView faces_infos)
195 {
196 
197  Real3 n0 = in_node_coord[fn_cv.nodeId(face_lid, 0)];
198  Real3 n1 = in_node_coord[fn_cv.nodeId(face_lid, 1)];
199  Real3 n2 = in_node_coord[fn_cv.nodeId(face_lid, 2)];
200 
201  if (!faces_infos.isSubDomainBoundaryOutside(face_lid))
202  std::swap(n0, n1);
203 
204  Real3 edge1 = { n1.x - n0.x, n1.y - n0.y, n1.z - n0.z };
205  Real3 edge2 = { n2.x - n0.x, n2.y - n0.y, n2.z - n0.z };
206 
207  Real3 normal = {
208  edge1.y * edge2.z - edge1.z * edge2.y,
209  edge1.z * edge2.x - edge1.x * edge2.z,
210  edge1.x * edge2.y - edge1.y * edge2.x
211  };
212 
213  Real norm = math::sqrt(normal.x * normal.x + normal.y * normal.y + normal.z * normal.z);
214  return { normal.x / norm, normal.y / norm, normal.z / norm };
215 }

314 ARCCORE_HOST_DEVICE static inline Quad4GaussPointInfo
315 computeGradientsAndJacobianQuad4(CellLocalId cell_lid,
316  const IndexedCellNodeConnectivityView& cn_cv,
317  const Accelerator::VariableNodeReal3InView& in_node_coord,
318  Real xi,
319  Real eta)
320 {
321  // Shape function derivatives ∂𝐍/∂ξ and ∂𝐍/∂η
322  // ∂𝐍/∂ξ = [ ∂C₁/∂ξ ∂𝑁₂/∂ξ ∂𝑁₃/∂ξ ∂𝑁₄/∂ξ ]
323  // ∂𝐍/∂η = [ ∂𝑁₁/∂η ∂𝑁₂/∂η ∂𝑁₃/∂η ∂𝑁₄/∂η ]
324  const Real dN_dxi[4] = { -0.25 * (1 - eta), 0.25 * (1 - eta), 0.25 * (1 + eta), -0.25 * (1 + eta) };
325  const Real dN_deta[4] = { -0.25 * (1 - xi), -0.25 * (1 + xi), 0.25 * (1 + xi), 0.25 * (1 - xi) };
326 
327  // Jacobian calculation 𝑱
328  // 𝑱 = [ 𝒋₀₀ 𝒋₀₁ ] = [ ∂𝑥/∂ξ ∂𝑦/∂ξ ]
329  // [ 𝒋₁₀ 𝒋₁₁ ] [ ∂𝑥/∂η ∂𝑦/∂η ]
330  //
331  // The Jacobian is computed as follows:
332  // 𝒋₀₀ = ∑ (∂𝑁ᵢ/∂ξ * 𝑥ᵢ) ∀ 𝑖= 𝟏,……,𝟒
333  // 𝒋₀₁ = ∑ (∂𝑁ᵢ/∂ξ * 𝑦ᵢ) ∀ 𝑖= 𝟏,……,𝟒
334  // 𝒋₁₀ = ∑ (∂𝑁ᵢ/∂η * 𝑦ᵢ) ∀ 𝑖= 𝟏,……,𝟒
335  // 𝒋₁₁ = ∑ (∂𝑁ᵢ/∂η * 𝑥ᵢ) ∀ 𝑖= 𝟏,……,𝟒
336  Real2x2 J;
337  for (Int8 a = 0; a < 4; ++a)
338  {
339  const auto& coord = in_node_coord[cn_cv.nodeId(cell_lid, a)];
340  J[0][0] += dN_dxi[a] * coord.x;
341  J[0][1] += dN_dxi[a] * coord.y;
342  J[1][0] += dN_deta[a] * coord.x;
343  J[1][1] += dN_deta[a] * coord.y;
344  }
345 
346  // Determinant of the Jacobian
347  const Real detJ = J[0][0] * J[1][1] - J[0][1] * J[1][0];
348 
349  if (detJ <= 0.0) {
350  //ARCANE_FATAL("Invalid (non-positive) Jacobian determinant: {0}", detJ);
351  }
352 
353  // Inverse of the Jacobian
354  // 𝑱⁻¹ = [ invJ00 invJ01 ]
355  // [ invJ10 invJ11 ]
356  const Real invJ00 = J[1][1] / detJ;
357  const Real invJ01 = -J[0][1] / detJ;
358  const Real invJ10 = -J[1][0] / detJ;
359  const Real invJ11 = J[0][0] / detJ;
360 
361  // Gradients in physical space (∂𝐍/∂𝑥, ∂𝐍/∂𝑦)
362  // {∂𝐍/∂𝑥} = [J]⁻¹ {∂𝐍/∂ξ}
363  // {∂𝐍/∂𝑦} {∂𝐍/∂η}
364  RealVector<4> dN_dx_result, dN_dy_result;
365  for (Int8 a = 0; a < 4; ++a) {
366  dN_dx_result(a) = invJ00 * dN_dxi[a] + invJ01 * dN_deta[a];
367  dN_dy_result(a) = invJ10 * dN_dxi[a] + invJ11 * dN_deta[a];
368  }
369 
370  return { dN_dx_result, dN_dy_result, detJ };
371 }

408 ARCCORE_HOST_DEVICE inline Real4
409 computeGradientXTetra4(CellLocalId cell_lid,
410  const IndexedCellNodeConnectivityView& cn_cv,
411  const Accelerator::VariableNodeReal3InView& in_node_coord)
412 {
413  Real3 n0 = in_node_coord[cn_cv.nodeId(cell_lid, 0)];
414  Real3 n1 = in_node_coord[cn_cv.nodeId(cell_lid, 1)];
415  Real3 n2 = in_node_coord[cn_cv.nodeId(cell_lid, 2)];
416  Real3 n3 = in_node_coord[cn_cv.nodeId(cell_lid, 3)];
417 
418  Real3 v0 = n1 - n0;
419  Real3 v1 = n2 - n0;
420  Real3 v2 = n3 - n0;
421 
422  // 6 x Volume of tetrahedron
423  Real V6 = std::abs(Arcane::math::dot(v0, Arcane::math::cross(v1, v2)));
424 
425  Real4 dx{};
426 
427  dx[0] = (n1.y * (n3.z - n2.z) + n2.y * (n1.z - n3.z) + n3.y * (n2.z - n1.z)) / V6;
428  dx[1] = (n0.y * (n2.z - n3.z) + n2.y * (n3.z - n0.z) + n3.y * (n0.z - n2.z)) / V6;
429  dx[2] = (n0.y * (n3.z - n1.z) + n1.y * (n0.z - n3.z) + n3.y * (n1.z - n0.z)) / V6;
430  dx[3] = (n0.y * (n1.z - n2.z) + n1.y * (n2.z - n0.z) + n2.y * (n0.z - n1.z)) / V6;
431 
432  return dx;
433 };

541 ARCCORE_HOST_DEVICE inline Hexa8GaussPointInfo
542 computeGradientsAndJacobianHexa8(CellLocalId cell_lid,
543  const IndexedCellNodeConnectivityView& cn_cv,
544  const Accelerator::VariableNodeReal3InView& in_node_coord,
545  Real xi, Real eta, Real zeta)
546 {
547  // Shape function derivatives ∂𝐍/∂ξ, ∂𝐍/∂η, ∂𝐍/∂ζ
548  // ∂𝐍/∂ξ = [ ∂𝑁₁/∂ξ ∂𝑁₂/∂ξ ∂𝑁₃/∂ξ ∂𝑁₄/∂ξ ∂𝑁₅/∂ξ ∂𝑁₆/∂ξ ∂𝑁₇/∂ξ ∂𝑁₈/∂ξ ]
549  // ∂𝐍/∂η = [ ∂𝑁₁/∂η ∂𝑁₂/∂η ∂𝑁₃/∂η ∂𝑁₄/∂η ∂𝑁₅/∂η ∂𝑁₆/∂η ∂𝑁₇/∂η ∂𝑁₈/∂η ]
550  // ∂𝐍/∂ζ = [ ∂𝑁₁/∂ζ ∂𝑁₂/∂ζ ∂𝑁₃/∂ζ ∂𝑁₄/∂ζ ∂𝑁₅/∂ζ ∂𝑁₆/∂ζ ∂𝑁₇/∂ζ ∂𝑁₈/∂ζ ]
551  Real dN_dxi[8], dN_deta[8], dN_dzeta[8];
552  const Real one_minus_eta = 1.0 - eta;
553  const Real one_plus_eta = 1.0 + eta;
554  const Real one_minus_xi = 1.0 - xi;
555  const Real one_plus_xi = 1.0 + xi;
556  const Real one_minus_zeta = 1.0 - zeta;
557  const Real one_plus_zeta = 1.0 + zeta;
558 
559  dN_dxi[0] = -0.125 * one_minus_eta * one_minus_zeta;
560  dN_dxi[1] = 0.125 * one_minus_eta * one_minus_zeta;
561  dN_dxi[2] = 0.125 * one_plus_eta * one_minus_zeta;
562  dN_dxi[3] = -0.125 * one_plus_eta * one_minus_zeta;
563  dN_dxi[4] = -0.125 * one_minus_eta * one_plus_zeta;
564  dN_dxi[5] = 0.125 * one_minus_eta * one_plus_zeta;
565  dN_dxi[6] = 0.125 * one_plus_eta * one_plus_zeta;
566  dN_dxi[7] = -0.125 * one_plus_eta * one_plus_zeta;
567 
568  dN_deta[0] = -0.125 * one_minus_xi * one_minus_zeta;
569  dN_deta[1] = -0.125 * one_plus_xi * one_minus_zeta;
570  dN_deta[2] = 0.125 * one_plus_xi * one_minus_zeta;
571  dN_deta[3] = 0.125 * one_minus_xi * one_minus_zeta;
572  dN_deta[4] = -0.125 * one_minus_xi * one_plus_zeta;
573  dN_deta[5] = -0.125 * one_plus_xi * one_plus_zeta;
574  dN_deta[6] = 0.125 * one_plus_xi * one_plus_zeta;
575  dN_deta[7] = 0.125 * one_minus_xi * one_plus_zeta;
576 
577  dN_dzeta[0] = -0.125 * one_minus_xi * one_minus_eta;
578  dN_dzeta[1] = -0.125 * one_plus_xi * one_minus_eta;
579  dN_dzeta[2] = -0.125 * one_plus_xi * one_plus_eta;
580  dN_dzeta[3] = -0.125 * one_minus_xi * one_plus_eta;
581  dN_dzeta[4] = 0.125 * one_minus_xi * one_minus_eta;
582  dN_dzeta[5] = 0.125 * one_plus_xi * one_minus_eta;
583  dN_dzeta[6] = 0.125 * one_plus_xi * one_plus_eta;
584  dN_dzeta[7] = 0.125 * one_minus_xi * one_plus_eta;
585 
586  // Jacobian matrix (default-initialized to zero see Real3x3.h)
587  // 𝑱 = [ 𝒋₀₀ 𝒋₀₁ 𝒋₀₂ ]
588  // [ 𝒋₁₀ 𝒋₁₁ 𝒋₁₂ ]
589  // [ 𝒋₂₀ 𝒋₂₁ 𝒋₂₂ ]
590  //
591  // The Jacobian is computed as follows:
592  // 𝒋₀₀ = ∑ (∂𝑥/∂ξ * 𝑥ᵢ) ∀ 𝑖= 𝟏,……,𝟖
593  // 𝒋₀₁ = ∑ (∂𝑥/∂ξ * 𝑦ᵢ) ∀ 𝑖= 𝟏,……,𝟖
594  // 𝒋₀₂ = ∑ (∂𝑥/∂ξ * 𝑧ᵢ) ∀ 𝑖= 𝟏,……,𝟖
595  // 𝒋₁₀ = ∑ (∂𝑦/∂η * 𝑥ᵢ) ∀ 𝑖= 𝟏,……,𝟖
596  // 𝒋₁₁ = ∑ (∂𝑦/∂η * 𝑦ᵢ) ∀ 𝑖= 𝟏,……,𝟖
597  // 𝒋₁₂ = ∑ (∂𝑦/∂η * 𝑧ᵢ) ∀ 𝑖= 𝟏,……,𝟖
598  // 𝒋₂₀ = ∑ (∂𝑧/∂ζ * 𝑥ᵢ) ∀ 𝑖= 𝟏,……,𝟖
599  // 𝒋₂₁ = ∑ (∂𝑧/∂ζ * 𝑦ᵢ) ∀ 𝑖= 𝟏,……,𝟖
600  // 𝒋₂₂ = ∑ (∂𝑧/∂ζ * 𝑧ᵢ) ∀ 𝑖= 𝟏,……,𝟖
601  Real3x3 J;
602  for (Int8 a = 0; a < 8; ++a) {
603  const Real3& n_coord = in_node_coord[cn_cv.nodeId(cell_lid, a)];
604  J[0][0] += dN_dxi[a] * n_coord.x; // ∂𝑥/∂ξ
605  J[0][1] += dN_dxi[a] * n_coord.y; // ∂𝑦/∂ξ
606  J[0][2] += dN_dxi[a] * n_coord.z; // ∂𝑧/∂ξ
607  J[1][0] += dN_deta[a] * n_coord.x; // ∂𝑥/∂η
608  J[1][1] += dN_deta[a] * n_coord.y; // ∂𝑦/∂η
609  J[1][2] += dN_deta[a] * n_coord.z; // ∂𝑧/∂η
610  J[2][0] += dN_dzeta[a] * n_coord.x; // ∂𝑥/∂ζ
611  J[2][1] += dN_dzeta[a] * n_coord.y; // ∂𝑦/∂ζ
612  J[2][2] += dN_dzeta[a] * n_coord.z; // ∂𝑧/∂ζ
613  }
614 
615  // Determinant and Inverse of the Jacobian
616  const Real detJ = math::matrixDeterminant(J);
617  if (detJ <= 0.0) {
618  //ARCANE_FATAL("Invalid (non-positive) Jacobian determinant: {0}", detJ);
619  }
620  const Real3x3 invJ = math::inverseMatrix(J, detJ);
621 
622  // Gradients in physical space (∂𝐍/∂𝑥, ∂𝐍/∂𝑦, ∂𝐍/∂𝑧)
623  // {∂𝐍/∂𝑥} = [J]⁻¹ {∂𝐍/∂ξ}
624  // {∂𝐍/∂𝑦} {∂𝐍/∂η}
625  // {∂𝐍/∂𝑧} {∂𝐍/∂ζ}
626  RealVector<8> dN_dx_result, dN_dy_result, dN_dz_result;
627  for (Int8 a = 0; a < 8; ++a) {
628  dN_dx_result(a) = invJ[0][0] * dN_dxi[a] + invJ[0][1] * dN_deta[a] + invJ[0][2] * dN_dzeta[a];
629  dN_dy_result(a) = invJ[1][0] * dN_dxi[a] + invJ[1][1] * dN_deta[a] + invJ[1][2] * dN_dzeta[a];
630  dN_dz_result(a) = invJ[2][0] * dN_dxi[a] + invJ[2][1] * dN_deta[a] + invJ[2][2] * dN_dzeta[a];
631  }
632 
633  return { dN_dx_result, dN_dy_result, dN_dz_result, detJ };
634 }
