277 _defineMatrixProfile(ProfilerT& profiler, ConstArrayView<Integer> allUIndex)
278 {
280  //
281  // DEFINE PROFILE
282  //
283  const Int32 dim = mesh()->dimension();
284  auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
285  ENUMERATE_ (Cell, icell, mesh()->allCells())
286  {
287  Cell cell = *icell;
288  for (Node node1 : cell.nodes())
289  {
290  if (node1.isOwn())
291  {
292  for (Node node2 : cell.nodes())
293  {
294  for (Int32 i = 0; i < dim; ++i)
295  {
296  DoFLocalId dof1 = node_dof.dofId(node1, i);
297  for (Int32 j = 0; j < dim; ++j)
298  {
299  DoFLocalId dof2 = node_dof.dofId(node2, j);
300  profiler.addMatrixEntry(allUIndex[dof1.localId()], allUIndex[dof2.localId()]);
301  }
302  }
303  }
304  }
305  }
306  }
