297 template <class Function> inline void BSRFormat::
298 assembleBilinearOrderedPerBlock(Function compute_element_matrix)
299 {
300  UnstructuredMeshConnectivityView m_connectivity_view(m_mesh);
301  auto cell_node_cv = m_connectivity_view.cellNode();
302 
303  ItemGenericInfoListView nodes_infos(m_mesh->nodeFamily());
304 
305  auto matrix_nb_row = m_bsr_matrix.nbRow();
306  auto matrix_nb_column = m_bsr_matrix.nbColumn();
307  auto matrix_nb_nz = m_bsr_matrix.nbNonZero();
308  auto matrix_nb_block = m_bsr_matrix.nbBlock();
309  auto matrix_nb_block_sq = matrix_nb_block * matrix_nb_block;
310 
311  auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
312 
313  auto command = makeCommand(m_queue);
314  auto in_row_index = viewIn(command, m_bsr_matrix._rowsIndex());
315  auto in_columns = viewIn(command, m_bsr_matrix._columns());
316  auto inout_values = viewInOut(command, m_bsr_matrix._values());
317  auto in_nz_per_row = viewIn(command, m_bsr_matrix._nbNonZeroPerRows());
318 
319  command << RUNCOMMAND_ENUMERATE(Cell, cell, m_mesh->allCells())
320  {
321  auto element_matrix = compute_element_matrix(cell);
322 
323  auto cur_row_node_idx = 0;
324  for (NodeLocalId row_node_lid : cell_node_cv.nodes(cell)) {
325  auto cur_col_node_idx = 0;
326  for (NodeLocalId col_node_lid : cell_node_cv.nodes(cell)) {
327  if (nodes_infos.isOwn(row_node_lid)) {
328  auto begin = in_row_index[row_node_lid];
329  auto end = (row_node_lid == matrix_nb_row - 1) ? matrix_nb_column : in_row_index[row_node_lid + 1];
330  while (begin < end) {
331  if (in_columns[begin] == col_node_lid) {
332  auto block_start = begin * matrix_nb_block_sq;
333  for (auto i = 0; i < matrix_nb_block; ++i) {
334  for (auto j = 0; j < matrix_nb_block; ++j) {
335  double value = element_matrix(matrix_nb_block * cur_row_node_idx + i, matrix_nb_block * cur_col_node_idx + j);
336  Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_values[block_start + (i * matrix_nb_block + j)], value);
337  }
338  }
339  break;
340  }
341  ++begin;
342  }
343  }
344  ++cur_col_node_idx;
345  }
346  ++cur_row_node_idx;
347  }
348  };
349 }

481 
482  auto cur_col_node_idx = 0;
483  for (NodeLocalId col_node_lid : cell_node_cv.nodes(cell)) {
484  if (nodes_infos.isOwn(row_node)) {
