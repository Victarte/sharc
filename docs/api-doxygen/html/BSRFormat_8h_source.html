256 template <class Function> inline void BSRFormat::
257 assembleBilinearOrderedPerBlock(Function compute_element_matrix)
258 {
259  UnstructuredMeshConnectivityView m_connectivity_view(m_mesh);
260  auto cell_node_cv = m_connectivity_view.cellNode();
261 
262  ItemGenericInfoListView nodes_infos(m_mesh->nodeFamily());
263 
264  auto matrix_nb_row = m_bsr_matrix.nbRow();
265  auto matrix_nb_column = m_bsr_matrix.nbColumn();
266  auto matrix_nb_nz = m_bsr_matrix.nbNonZero();
267  auto matrix_nb_block = m_bsr_matrix.nbBlock();
268  auto matrix_nb_block_sq = matrix_nb_block * matrix_nb_block;
269 
270  auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
271 
272  auto command = makeCommand(m_queue);
273  auto in_row_index = viewIn(command, m_bsr_matrix._rowsIndex());
274  auto in_columns = viewIn(command, m_bsr_matrix._columns());
275  auto inout_values = viewInOut(command, m_bsr_matrix._values());
276  auto in_nz_per_row = viewIn(command, m_bsr_matrix._nbNonZeroPerRows());
277 
278  command << RUNCOMMAND_ENUMERATE(Cell, cell, m_mesh->allCells())
279  {
280  auto element_matrix = compute_element_matrix(cell);
281 
282  auto cur_row_node_idx = 0;
283  for (NodeLocalId row_node_lid : cell_node_cv.nodes(cell)) {
284  auto cur_col_node_idx = 0;
285  for (NodeLocalId col_node_lid : cell_node_cv.nodes(cell)) {
286  if (nodes_infos.isOwn(row_node_lid)) {
287  auto begin = in_row_index[row_node_lid];
288  auto end = (row_node_lid == matrix_nb_row - 1) ? matrix_nb_column : in_row_index[row_node_lid + 1];
289  while (begin < end) {
290  if (in_columns[begin] == col_node_lid) {
291  auto block_start = begin * matrix_nb_block_sq;
292  for (auto i = 0; i < matrix_nb_block; ++i) {
293  for (auto j = 0; j < matrix_nb_block; ++j) {
294  double value = element_matrix(matrix_nb_block * cur_row_node_idx + i, matrix_nb_block * cur_col_node_idx + j);
295  Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_values[block_start + (i * matrix_nb_block + j)], value);
296  }
297  }
298  break;
299  }
300  ++begin;
301  }
302  }
303  ++cur_col_node_idx;
304  }
305  ++cur_row_node_idx;
306  }
307  };
308 }

410  auto node_cell_cv = m_connectivity_view.nodeCell();

412  ItemGenericInfoListView nodes_infos(m_mesh->nodeFamily());

435  }
436  cur_row_node_idx = 0;
437  }
438 
439  auto element_vector = compute_element_vectors(cell, cur_row_node_idx);
440 
441  auto cur_col_node_idx = 0;
442  for (NodeLocalId col_node_lid : cell_node_cv.nodes(cell)) {
443  if (nodes_infos.isOwn(row_node)) {
444 
445  auto begin = in_row_index[row_node];
446  auto end = (row_node == matrix_nb_row - 1) ? matrix_nb_column : in_row_index[row_node + 1];
447

450  auto block_start = begin * matrix_nb_block_sq;
451 
452  for (auto i = 0; i < matrix_nb_block; ++i) {
453  for (auto j = 0; j < matrix_nb_block; ++j) {
454  double value = element_vector(i, matrix_nb_block * cur_col_node_idx + j);

459  }
