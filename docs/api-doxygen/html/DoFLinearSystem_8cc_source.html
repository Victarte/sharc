116  void solve() override
117  {
118 
119  Int32 matrix_size = m_k_matrix.extent0();
120  Arcane::MatVec::Matrix matrix(matrix_size, matrix_size);
121  _convertNumArrayToCSRMatrix(matrix, m_k_matrix.mdspan());
122  bool is_verbose = true;
123  Arcane::MatVec::Vector vector_b(matrix_size);
124  Arcane::MatVec::Vector vector_x(matrix_size);
125  {
126  auto vector_b_view = vector_b.values();
127  auto vector_x_view = vector_x.values();
128  for (Int32 i = 0; i < matrix_size; ++i) {
129  vector_b_view(i) = m_rhs_vector[i];
130  if (is_verbose)
131  info() << "VectorB[" << i << "] = " << m_rhs_vector[i];
132  vector_x_view(i) = 0.0;
133  }
134  }
135 
136  bool use_direct_solver = false;
137  switch (m_solver_method) {
138  case eInternalSolverMethod::Auto:
139  use_direct_solver = matrix_size < 500;
140  break;
141  case eInternalSolverMethod::Direct:
142  use_direct_solver = true;
143  break;
144  case eInternalSolverMethod::PCG:
145  use_direct_solver = false;
146  break;
147  }
148  if (use_direct_solver) {
149  info() << "Using direct solver";
150  Arcane::MatVec::DirectSolver solver;
151  solver.solve(matrix, vector_b, vector_x);
152  }
153  else {
154  Real epsilon = m_epsilon;
155  info() << "Using internal solver with diagonal preconditioner epsilon=" << epsilon;
156  Arcane::MatVec::DiagonalPreconditioner p(matrix);
157  Arcane::MatVec::ConjugateGradientSolver solver;
158  solver.solve(matrix, vector_b, vector_x, epsilon, &p);
159  info() << "End solver nb_iteration=" << solver.nbIteration()
160  << " residual_norm=" << solver.residualNorm();
161  }
162 
163  {
164  auto vector_x_view = vector_x.values();
165  ENUMERATE_ (DoF, idof, m_dof_family->allItems()) {
166  DoF dof = *idof;
167  Real v = vector_x_view[dof.localId()];
168  m_dof_variable[dof] = v;
169  if (is_verbose)
170  info() << "VectorX[" << dof.localId() << "] = " << v;
171  }
172  }
173  }

270  void _fillRHSVector()
271  {
272  // For the LinearSystem class we need an array
273  // with only the values for the ownNodes().
274  // The values of 'rhs_values' should not be updated after
275  // this call.
276  UniqueArray<Real> rhs_values_for_linear_system;
277  VariableDoFReal& rhs_values(rhsVariable());
278  ENUMERATE_ (DoF, idof, m_dof_family->allItems().own()) {
279  rhs_values_for_linear_system.add(rhs_values[idof]);
280  }
281  _setRHSValues(rhs_values_for_linear_system);
282  }
283 
284  void _setRHSValues(Span<const Real> values)
285  {
286  Int32 index = 0;
287  DoFGroup own_dofs = m_dof_family->allItems().own();
288  ENUMERATE_ (DoF, idof, own_dofs) {
289  DoFLocalId dof_id = *idof;
290  m_rhs_vector[dof_id] = values[index];
291  ++index;
292  }
293  }

307 class DefaultDoFLinearSystemFactory
308 : public IDoFLinearSystemFactory
309 {
310  public:
311 
312  IDoFLinearSystemImpl* createInstance(ISubDomain* sd, IItemFamily* dof_family,
313  const String& solver_name) override
314  {
315  IParallelMng* pm = sd->parallelMng();
316  bool is_parallel = pm->isParallel();
317  // If true, we use a dense debug matrix in sequential
318  bool use_debug_dense_matrix = false;
319 #ifdef ENABLE_DEBUG_MATRIX
320  use_debug_dense_matrix = true;
321 #endif
322  IDoFLinearSystemImpl* p = nullptr;
323  if (is_parallel || !use_debug_dense_matrix) {
324  p = createAlephDoFLinearSystemImpl(sd, dof_family, solver_name);
325  }
326  else {
327  auto* x = new SequentialDoFLinearSystemImpl(sd, dof_family, solver_name);
328  x->build();
329  p = x;
330  }
331  return p;
332  }
333 };

369 void DoFLinearSystem::
370 initialize(ISubDomain* sd, Runner* runner, IItemFamily* dof_family, const String& solver_name)
371 {
372  ARCANE_CHECK_POINTER(sd);
373  ARCANE_CHECK_POINTER(dof_family);
374  ITraceMng* tm = dof_family->traceMng();
375  if (m_p)
376  ARCANE_FATAL("The instance is already initialized");
377  if (!m_linear_system_factory) {
378  tm->info() << "Creating default Linear System Factory";
379  if (!m_default_linear_system_factory)
380  m_default_linear_system_factory = new DefaultDoFLinearSystemFactory();
381  m_linear_system_factory = m_default_linear_system_factory;
382  }
383  m_item_family = dof_family;
384  m_p = m_linear_system_factory->createInstance(sd, dof_family, solver_name);
385  if (runner)
386  m_p->setRunner(*runner);
387 }
