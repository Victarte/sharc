48 void BSRMatrix::
49 initialize(Int32 nb_non_zero_value, Int32 nb_col, Int32 nb_row, Int8 nb_block, bool order_values_per_block)
50 {
51  if (nb_block <= 0 || nb_non_zero_value <= 0 || nb_row <= 0)
52  ARCANE_THROW(ArgumentException, "BSRMatrix(initialize): arguments should be positive and not null (block_size={0}, nb_non_zero_value={1} and nb_row={2})", nb_block, nb_non_zero_value, nb_row);
53 
54  if (nb_block > nb_row)
55  ARCANE_THROW(ArgumentException, "BSRMatrix(initialize): block_size should be less than nb_row");
56 
57  info() << "BSRMatrix(initialize): Initialize BSRMatrix with block_size=" << nb_block
58  << ", nb_non_zero_value=" << nb_non_zero_value
59  << ", nb_col=" << nb_col << ", nb_row=" << nb_row
60  << ", order_values_per_block=" << std::boolalpha << order_values_per_block;
61 
62  m_order_values_per_block = order_values_per_block;
63 
64  m_nb_non_zero_value = nb_non_zero_value;
65  m_nb_col = nb_col;
66  m_nb_row = nb_row;
67  m_nb_block = nb_block;
68 
69  m_values.resize(nb_non_zero_value);
70  m_values.fill(0, &m_queue);
71  m_columns.resize(nb_col);
72  m_rows_index.resize(nb_row);
73  m_nb_non_zero_per_rows.resize(nb_row);
74 }

79 Int32 BSRMatrix::
80 findValueIndex(DoFLocalId row, DoFLocalId col) const
81 {
82  auto block_row = row / m_nb_block;
83  auto block_col = col / m_nb_block;
84 
85  auto block_start = m_rows_index[block_row];
86  auto block_end = (block_row == m_nb_row - 1) ? m_nb_col : m_rows_index[block_row + 1];
87 
88  auto row_offset = row % m_nb_block;
89  auto col_offset = col % m_nb_block;
90 
91  auto nb_block_sq = m_nb_block * m_nb_block;
92  auto block_start_in_value = block_start * nb_block_sq;
93  auto col_index = 0;
94  while (block_start < block_end) {
95  if (m_columns[block_start] == block_col) {
96  if (!m_order_values_per_block)
97  return block_start_in_value + (m_nb_block * col_index) + (row_offset * m_nb_block * m_nb_non_zero_per_rows[block_row]) + col_offset;
98  else
99  return (block_start * nb_block_sq) + ((row_offset * m_nb_block) + col_offset);
100  }
101  ++block_start;
102  ++col_index;
103  }
104 
105  ARCANE_FATAL("BSRMatrix(findValueIndex): Value not found");
106 }

110 void BSRMatrix::
111 toCsr(CsrFormat* csr_matrix)
112 {
113  info() << "BSRMatrix(toCsr): Convert matrix to CSR";
114  auto startTime = platform::getRealTime();
115 
116  auto nb_block_rows = m_rows_index.extent0();
117  auto nb_rows = (nb_block_rows * m_nb_block) + 1;
118  auto total_non_zero_elements = m_nb_non_zero_value;
119 
120  csr_matrix->initialize(nullptr, total_non_zero_elements, nb_rows, m_queue);
121 
122  if (m_nb_block == 1) {
123  csr_matrix->m_matrix_row = m_rows_index;
124  csr_matrix->m_matrix_column = m_columns;
125  csr_matrix->m_matrix_rows_nb_column = m_nb_non_zero_per_rows;
126  }
127  else {
128  // Translate `row_index`
129  csr_matrix->m_matrix_row[0] = 0;
130  auto offset = 1;
131  for (auto i = 0; i < m_rows_index.extent0(); ++i) {
132  for (auto j = 0; j < m_nb_block; ++j) {
133  auto start = m_rows_index[i];
134  auto end = i == m_rows_index.extent0() - 1 ? m_nb_col : m_rows_index[i + 1];
135  auto nombre_de_dof_dans_la_rangee = (end - start) * m_nb_block;
136  csr_matrix->m_matrix_row[offset] = csr_matrix->m_matrix_row[offset - 1] + nombre_de_dof_dans_la_rangee;
137  offset++;
138  }
139  }
140 
141  // Translate `columns`
142  offset = 0;
143  for (auto i = 0; i < m_rows_index.extent0(); ++i) {
144  for (auto j = 0; j < m_nb_block; ++j) {
145  auto start = m_rows_index[i];
146  auto end = i == m_rows_index.extent0() - 1 ? m_nb_col : m_rows_index[i + 1];
147  for (auto block_index = start; block_index < end; ++block_index) {
148  for (auto k = 0; k < m_nb_block; ++k) {
149  csr_matrix->m_matrix_column[offset] = m_columns[block_index] * m_nb_block + k;
150  offset++;
151  }
152  }
153  }
154  }
155 
156  // Translate `nb_nz_per_row`
157  offset = 0;
158  for (auto i = 0; i < m_nb_non_zero_per_rows.extent0(); ++i) {
159  for (auto j = 0; j < m_nb_block; ++j) {
160  csr_matrix->m_matrix_rows_nb_column[offset++] = m_nb_non_zero_per_rows[i] * m_nb_block;
161  }
162  }
163 
164  // NOTE: The last entry calculation might not be needed now, but keep for safety
165  // auto row_index_size = csr_matrix->m_matrix_row.extent0();
166  // csr_matrix->m_matrix_row[row_index_size - 1] = total_non_zero_elements;
167  }
168 
169  // NOTE: If we don't want to keep bsr matrix coefficients we could move the data instead of copying it.
170  csr_matrix->m_matrix_value = m_values;
171  info() << std::left << std::setw(40) << "[BsrMatrix-Timer] convert-bsr-to-csr" << " = " << (platform::getRealTime() - startTime);
172 }

177 void BSRMatrix::
178 toLinearSystem(DoFLinearSystem& linear_system)
179 {
180  info() << "BSRMatrix(toLinearSystem): Translate matrix to linear system using `matrixAddValue`";
181 
182  for (auto row = 0; row < m_nb_row; ++row) {
183  auto row_start = m_rows_index[row];
184  auto row_end = (row + 1 < m_nb_row) ? m_rows_index[row + 1] : m_nb_col;
185  for (auto block_idx = row_start; block_idx < row_end; ++block_idx) {
186  auto col = m_columns[block_idx];
187  for (auto i = 0; i < m_nb_block; ++i) {
188  for (auto j = 0; j < m_nb_block; ++j) {
189  auto global_row = (row * m_nb_block) + i;
190  auto global_col = (col * m_nb_block) + j;
191  auto value = getValue(DoFLocalId(global_row), DoFLocalId(global_col));
192  linear_system.matrixAddValue(DoFLocalId(global_row), DoFLocalId(global_col), value);
193  }
194  }
195  }
196  }

242 Int64 BSRFormat::
243 computeNbEdge(IMesh* mesh)
244 {
245  Int64 nb_edge = 0;
246  if (mesh->dimension() == 2)
247  nb_edge = mesh->nbFace();
248  else {
249  auto nn_via_edge_cv = MeshUtils::computeNodeNodeViaEdgeConnectivity(mesh, "NodeNodeViaEdge");
250  IndexedNodeNodeConnectivityView nn_cv = nn_via_edge_cv->view();
251  ENUMERATE_NODE (inode, mesh->allNodes()) {
252  Node node = *inode;
253  nb_edge += nn_cv.nbNode(node);
254  }
255  nb_edge /= 2;
256  }
257  return nb_edge;
258 }

263 void BSRFormat::
264 initialize(IMesh* mesh, Int8 nb_dof, bool does_linear_system_use_csr, bool use_atomic_free)
265 {
266  ARCANE_CHECK_POINTER(mesh);
267 
268  if (mesh->dimension() != 2 && mesh->dimension() != 3)
269  ARCANE_THROW(NotImplementedException, "BSRFormat(initialize): Only supports 2D and 3D");
270 
271  auto startTime = platform::getRealTime();
272 
273  m_mesh = mesh;
274  m_nb_dof = nb_dof;
275  Int64 nb_edge = computeNbEdge(mesh);
276  Int32 nb_node = m_mesh->nbNode();
277  Int32 nb_col = 2 * nb_edge + nb_node;
278  Int32 nb_non_zero_value = (m_nb_dof * m_nb_dof) * (2 * nb_edge + nb_node);
279 
280  m_use_csr_in_linear_system = does_linear_system_use_csr;
281  bool order_values_per_block = !does_linear_system_use_csr;
282  m_bsr_matrix.initialize(nb_non_zero_value, nb_col, nb_node, m_nb_dof, order_values_per_block);
283  m_use_atomic_free = use_atomic_free;
284 
285  info() << std::left << std::setw(40) << "[BsrMatrix-Timer] initialize-bsr-matrix" << " = " << (platform::getRealTime() - startTime);
286 }

314 void BSRFormat::
315 computeNzPerRowArray()
316 {
317  info() << "BSRFormat(computeNzPerRowArray): Compute nb_nz_per_row BSR matrix array";
318  auto startTime = platform::getRealTime();
319 
320  auto command = makeCommand(m_queue);
321  NumArray<Int32, MDDim1>& nb_nz_per_row = m_bsr_matrix._nbNonZeroPerRows();
322  auto inout_nb_nz_per_row = viewInOut(command, nb_nz_per_row);
323  auto inout_row_index = viewInOut(command, m_bsr_matrix._rowsIndex());
324 
325  auto nb_row = m_bsr_matrix.nbRow();
326  // Copy `row_index` in `nz_per_row`
327  // NOTE: Does Arcane provide a copy-on-gpu primitive ?
328  {
329  auto command = makeCommand(m_queue);
330  command << RUNCOMMAND_LOOP1(iter, nb_row)
331  {
332  auto [i] = iter();
333  inout_nb_nz_per_row[i] = inout_row_index[i];
334  };
335  }
336  m_queue.barrier();
337 
338  {
339  auto nb_col = m_bsr_matrix.nbColumn();
340  auto command = makeCommand(m_queue);
341  command << RUNCOMMAND_LOOP1(iter, nb_row)
342  {
343  auto [i] = iter();
344  auto x = inout_nb_nz_per_row[i];
345  if (i == nb_row - 1)
346  inout_nb_nz_per_row[i] = nb_col - inout_row_index[i];
347  else
348  inout_nb_nz_per_row[i] = inout_row_index[i + 1] - x;
349  };
350  }
351  m_queue.barrier();
352 
353  info() << std::left << std::setw(40) << "[BsrMatrix-Timer] compute-nnz-bsr" << " = " << (platform::getRealTime() - startTime);
354 }

402 void BSRFormat::
403 computeColumnsAtomicFree()
404 {
405  auto command = makeCommand(m_queue);
406  auto inout_columns = viewInOut(command, m_bsr_matrix._columns());
407  auto row_index = m_bsr_matrix._rowsIndex().to1DSmallSpan();
408 
409  if (m_mesh->dimension() == 2) {
410  UnstructuredMeshConnectivityView connectivity_view(m_mesh);
411  auto node_face_cv = connectivity_view.nodeFace();
412  auto face_node_cv = connectivity_view.faceNode();
413 
414  command << RUNCOMMAND_ENUMERATE(Node, node_id, m_mesh->allNodes())
415  {
416  auto offset = row_index[node_id];
417 
418  for (auto face_lid : node_face_cv.faceIds(node_id)) {
419  auto nodes = face_node_cv.nodes(face_lid);
420  inout_columns[offset] = nodes[0] == node_id ? nodes[1] : nodes[0];
421  ++offset;
422  }
423 
424  inout_columns[offset] = node_id;
425  };
426  }
427  else {
428  auto connectivity_mng = m_mesh->indexedConnectivityMng();
429  auto connectivity_ptr = connectivity_mng->findOrCreateConnectivity(m_mesh->nodeFamily(), m_mesh->nodeFamily(), "NodeNodeViaEdge");
430  IndexedNodeNodeConnectivityView node_node_cv = connectivity_ptr->view();
431 
432  command << RUNCOMMAND_ENUMERATE(Node, node_id, m_mesh->allNodes())
433  {
434  auto offset = row_index[node_id];
435 
436  for (auto neighbor_idx : node_node_cv.nodeIds(node_id)) {
437  inout_columns[offset] = neighbor_idx;
438  ++offset;
439  }
440 
441  inout_columns[offset] = node_id;
442  };
443  }

467 void BSRFormat::
468 computeSortedEdges(Int8 edges_per_element, Int64 nb_edge_total, SmallSpan<UInt64>& sorted_edges_ss)
469 {
470  auto mem_ressource = m_queue.memoryRessource();
471  NumArray<UInt64, MDDim1> edges(mem_ressource);
472  edges.resize(nb_edge_total);
473 
474  UnstructuredMeshConnectivityView m_connectivity_view(m_mesh);
475  auto cell_node_cv = m_connectivity_view.cellNode();
476 
477  {
478  auto command = makeCommand(m_queue);
479  auto inout_edges = viewInOut(command, edges);
480 
481  if (m_mesh->dimension() == 2) {
482  command << RUNCOMMAND_ENUMERATE(CellLocalId, cell_lid, m_mesh->allCells())
483  {
484  auto n0 = cell_node_cv.nodeId(cell_lid, 0);
485  auto n1 = cell_node_cv.nodeId(cell_lid, 1);
486  auto n2 = cell_node_cv.nodeId(cell_lid, 2);
487 
488  auto start = cell_lid * edges_per_element;
489  inout_edges[start] = pack(n0, n1);
490  inout_edges[start + 1] = pack(n0, n2);
491  inout_edges[start + 2] = pack(n1, n2);
492  };
493  }
494  else {
495  command << RUNCOMMAND_ENUMERATE(CellLocalId, cell_lid, m_mesh->allCells())
496  {
497  auto n0 = cell_node_cv.nodeId(cell_lid, 0);
498  auto n1 = cell_node_cv.nodeId(cell_lid, 1);
499  auto n2 = cell_node_cv.nodeId(cell_lid, 2);
500  auto n3 = cell_node_cv.nodeId(cell_lid, 3);
501 
502  auto start = cell_lid * edges_per_element;
503  inout_edges[start] = pack(n0, n1);
504  inout_edges[start + 1] = pack(n0, n2);
505  inout_edges[start + 2] = pack(n0, n3);
506  inout_edges[start + 3] = pack(n1, n2);
507  inout_edges[start + 4] = pack(n1, n3);
508  inout_edges[start + 5] = pack(n2, n3);
509  };
510  }
511  }
512  m_queue.barrier();
513 
514  auto sorter = Accelerator::GenericSorter(m_queue);
515  SmallSpan<const UInt64> edges_ss = edges.to1DSmallSpan();
516  sorter.apply(edges_ss, sorted_edges_ss);
517 }

522 void BSRFormat::
523 computeNeighbors(Int8 edges_per_element, Int64 nb_edge_total, NumArray<Int32, MDDim1>& neighbors, SmallSpan<UInt64>& sorted_edges_ss)
524 {
525  auto command = makeCommand(m_queue);
526  auto inout_neighbors = viewInOut(command, neighbors);
527 
528  command << RUNCOMMAND_LOOP1(iter, nb_edge_total)
529  {
530  auto [thread_id] = iter();
531  auto cur_edge = sorted_edges_ss[thread_id];
532  if (thread_id == (nb_edge_total - 1) || cur_edge != sorted_edges_ss[thread_id + 1]) {
533  Int32 n0, n1 = 0;
534  unpack(cur_edge, n0, n1);
535  Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_neighbors[n0], 1);
536  Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_neighbors[n1], 1);
537  }
538  };
539 }

561 void BSRFormat::
562 computeColumns(Int8 edges_per_element, Int64 nb_edge_total, SmallSpan<uint64_t>& sorted_edges_ss)
563 {

566  {

569  auto inout_columns = viewInOut(command, m_bsr_matrix._columns());
570 
571  command << RUNCOMMAND_LOOP1(iter, nb_node)
572  {
573  auto [n] = iter();
574  auto start = in_row_index[n];
575  inout_columns[start] = n;
576  };
577  }
578  m_queue.barrier();
579 
580  auto mem_ressource = m_queue.memoryRessource();
581  NumArray<Int32, MDDim1> offsets(mem_ressource);
582  offsets.resize(nb_node);
583  offsets.fill(1, &m_queue);
584 
585  {
586  auto command = makeCommand(m_queue);
587  auto inout_columns = viewInOut(command, m_bsr_matrix._columns());
588  auto in_row_index = viewIn(command, m_bsr_matrix._rowsIndex());
589  auto inout_offsets = viewInOut(command, offsets);
590 
591  command << RUNCOMMAND_LOOP1(iter, nb_edge_total)
592  {
593  auto [thread_id] = iter();
594  auto cur_edge = sorted_edges_ss[thread_id];
595  if (thread_id == (nb_edge_total - 1) || cur_edge != sorted_edges_ss[thread_id + 1]) {
596  Int32 n0, n1 = 0;
597  unpack(cur_edge, n0, n1);
598  registerEdgeInColumns(n0, n1, inout_offsets, in_row_index, inout_columns);
599  registerEdgeInColumns(n1, n0, inout_offsets, in_row_index, inout_columns);
600  }
601  };
602  }
603 };
