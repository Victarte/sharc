71 void FemDoFsOnNodes::Impl::
72 initialize(IMesh* mesh, Int32 nb_dof_per_node)
73 {
74  IItemFamily* dof_family_interface = mesh->findItemFamily(Arcane::IK_DoF, "DoFNodeFamily", true);
75  mesh::DoFFamily* dof_family = ARCANE_CHECK_POINTER(dynamic_cast<mesh::DoFFamily*>(dof_family_interface));
76  m_dof_family = dof_family_interface;
77 
78  // Create the DoFs
79  Int64UniqueArray uids(mesh->allNodes().size() * nb_dof_per_node);
80  {
81  Integer dof_index = 0;
82  // Use a mask to make sure the uniqueId() of the dof
83  // can not be negative if we multiply the uniqueId().
84  const UInt64 uid_mask = (1<<28) - 1;
85  ENUMERATE_NODE (inode, mesh->allNodes()) {
86  Node node = *inode;
87  Int64 node_unique_id = node.uniqueId().asInt64();
88  for (Integer i = 0; i < nb_dof_per_node; ++i) {
89  uids[dof_index] = (node_unique_id & uid_mask) * nb_dof_per_node + i;
90  ++dof_index;
91  }
92  }
93  }
94  //info() << "ADD_Dofs list=" << uids;
95  Int32UniqueArray dof_lids(uids.size());
96  dof_family->addDoFs(uids, dof_lids);
97  dof_family->endUpdate();
98  info() << "NB_DOF=" << dof_family->allItems().size();
99 
100  // Create Node -> DoF connectivity.
101  m_node_dof_connectivity = mesh->indexedConnectivityMng()->findOrCreateConnectivity(mesh->nodeFamily(), dof_family, "DoFNode");
102  auto* cn = m_node_dof_connectivity->connectivity();
103  {
104  Integer dof_index = 0;
105  ENUMERATE_NODE (inode, mesh->allNodes()) {
106  NodeLocalId node = *inode;
107  for (Integer i = 0; i < nb_dof_per_node; ++i) {
108  cn->addConnectedItem(node, DoFLocalId(dof_lids[dof_index]));
109  ++dof_index;
110  }
111  }
112  }
113  info() << "End build Dofs";
114 
115  IndexedNodeDoFConnectivityView node_dof(m_node_dof_connectivity->view());
116  {
117  // Set the owners of the DoF.
118  IParallelMng* pm = mesh->parallelMng();
119  Int32 my_rank = pm->commRank();
120  ItemInternalList dofs = m_dof_family->itemsInternal();
121  ENUMERATE_ (Node, inode, mesh->allNodes()) {
122  Node node = *inode;
123  Int32 node_owner = node.owner();
124  for (DoFLocalId dof : node_dof.dofs(node)) {
125  dofs[dof]->setOwner(node_owner, my_rank);
126  }
127  }
128  dof_family->notifyItemsOwnerChanged();
129  dof_family->computeSynchronizeInfos();
130  }
131 }
