23 void RelativePermeabilityPowerLaw::eval( const Real Sw, Real& krw, Real& dkrw_dSw ) const
24 {
25  // Compute results
26  Real dSwBar_dSw;
27  const Real SwBar = computeSwBar( Sw, dSwBar_dSw );
28  krw = pow(SwBar, m_alpha);

35 Real RelativePermeabilityPowerLaw::computeSwBar( const Real Sw, Real& dSwBar_dSw ) const
36 {
37  // Compute results
38  Real SwBar;
39  if ( Sw < m_Swi )
40  {
41  SwBar = 0;
42  dSwBar_dSw = 0;
43  }
44  else if ( Sw <= 1 - m_Sgc )
45  {
46  SwBar = ( Sw - m_Swi ) / ( 1 - m_Sgc - m_Swi );
47  dSwBar_dSw = 1 / ( 1 - m_Sgc - m_Swi );
48  }
49  else
50  {
51  SwBar = 1;
52  dSwBar_dSw = 0;
53  }
54  return SwBar;
55 }
