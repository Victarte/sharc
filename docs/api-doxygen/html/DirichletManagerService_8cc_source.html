131 DirichletManagerService::
132 init(ArcRes::System& system)
133 {
134  if(m_initialized) return;
135 
136  _checkSizeMulti();
137 
138  for(Arcane::Integer i = 0; i < options()->boundary.size(); ++i) {
139 
140  auto model_name = Arcane::String::format("Dirichlet_{0}",i);
141 
142  const Arcane::String face_group_name = options()->boundary[i].faceGroup();
143  ItemGroup group = mesh()->faceFamily()->findGroup(face_group_name);
144  if(group.null()) {
145  fatal() << "Dirichlet model cell zone "<<face_group_name<<" not exist";
146  }
147  auto name = Arcane::String::format("{0}_{1}",model_name,group.name());
148  auto* folder = new Law::PartialVariableFaceFolder(name,mesh());
149  m_boundaries.add(folder);
150  folder->setSupport(group);
151  if(options()->boundary[i]->name.isPresent()) {
152  m_boundary_names.add(options()->boundary[i]->name());
153  } else {
154  m_boundary_names.add(model_name);
155  }
156 
157  //==================
158  // Add Darcy Velocity
159  //==================
160 
161  ENUMERATE_PHASE(iphase, system.fluidSubSystem().phases()) {
162  const ArcRes::Phase & phase = *iphase;
163  folder->addVariable(ArcRes::DarcyVelocity(phase));
164  }
165 
166  //==================
167  // Space-time
168  //==================
169 
170  for(Arcane::Integer j = 0; j < options()->boundary[i].spaceTimeLimitCondition.size(); ++j) {
171  auto& condition(options()->boundary[i].spaceTimeLimitCondition[j]) ;
172  auto p = ArcRes::XPath::scalarRealProperty(system, condition.property());
173  folder->addVariable(p);
174  condition->function()->init() ;
175  }
176 
177  //==================
178  // Single
179  //==================
180 
181  for(Arcane::Integer j = 0; j < options()->boundary[i].limitCondition.size(); ++j) {
182  auto p = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i].limitCondition[j].property());
183  folder->addVariable(p);
184  }
185 
186  //==================
187  // Multi
188  //==================
189 
190  for(Arcane::Integer j = 0; j < options()->boundary[i].multiLimitCondition.size(); ++j) {
191 
192  Arcane::Integer num_prop = options()->boundary[i]->multiLimitCondition[j]->property.size();
193 
194  for ( Arcane::Integer iprop = 0 ; iprop < num_prop; iprop++ )
195  {
196  auto p = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i].multiLimitCondition[j]->property[iprop]);
197  folder->addVariable(p);
198  }
199  }
200  }
201 
202  _compute(system);
203 
204  m_initialized = true;
205 }
