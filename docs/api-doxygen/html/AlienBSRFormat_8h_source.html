305 template <class Function> inline void AlienBSRFormat::
306 assembleBilinearOrderedPerBlock(Function compute_element_matrix)
307 {
308  UnstructuredMeshConnectivityView m_connectivity_view(m_mesh);
309  auto cell_node_cv = m_connectivity_view.cellNode();
310 
311  ItemGenericInfoListView nodes_infos(m_mesh->nodeFamily());
312 
313  auto matrix_nb_row = m_bsr_matrix.nbRow();
314  auto matrix_nb_column = m_bsr_matrix.nbColumn();
315  //auto matrix_nb_nz = m_bsr_matrix.nbNonZero();
316  auto matrix_nb_block = m_bsr_matrix.nbBlock();
317  auto matrix_nb_block_sq = matrix_nb_block * matrix_nb_block;
318 
319  //auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
320 
321  auto command = makeCommand(m_queue);
322  auto in_row_index = viewIn(command, m_bsr_matrix._rowsIndex());
323  auto in_columns = viewIn(command, m_bsr_matrix._columns());
324  auto inout_values = viewInOut(command, m_bsr_matrix._values());
325  auto in_nz_per_row = viewIn(command, m_bsr_matrix._nbNonZeroPerRows());
326 
327  command << RUNCOMMAND_ENUMERATE(Cell, cell, m_mesh->allCells())
328  {
329  auto element_matrix = compute_element_matrix(cell);
330 
331  auto cur_row_node_idx = 0;
332  for (NodeLocalId row_node_lid : cell_node_cv.nodes(cell)) {
333  auto cur_col_node_idx = 0;
334  for (NodeLocalId col_node_lid : cell_node_cv.nodes(cell)) {
335  if (nodes_infos.isOwn(row_node_lid)) {
336  auto begin = in_row_index[row_node_lid];
337  auto end = (row_node_lid == matrix_nb_row - 1) ? matrix_nb_column : in_row_index[row_node_lid + 1];
338  while (begin < end) {
339  if (in_columns[begin] == col_node_lid) {
340  auto block_start = begin * matrix_nb_block_sq;
341  for (auto i = 0; i < matrix_nb_block; ++i) {
342  for (auto j = 0; j < matrix_nb_block; ++j) {
343  double value = element_matrix(matrix_nb_block * cur_row_node_idx + i, matrix_nb_block * cur_col_node_idx + j);
344  Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_values[block_start + (i * matrix_nb_block + j)], value);
345  }
346  }
347  break;
348  }
349  ++begin;
350  }
351  }
352  ++cur_col_node_idx;
353  }
354  ++cur_row_node_idx;
355  }
356  };
357 }
