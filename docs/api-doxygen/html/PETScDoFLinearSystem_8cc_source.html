172 void PETScDoFLinearSystemImpl::
173 _handleParameters(IParallelMng* pm)
174 {
175  PetscBool is_initialized;
176  PetscInitialized(&is_initialized);
177 
178  if (!is_initialized) // no petsc_flags were given
179  PetscInitialize(nullptr, nullptr, nullptr, nullptr);
180 
181 #define MAX_STRING_LENGTH 256
182 
183 #define PETSC_OPTION(type, petsc_string, variable) \
184  PetscOptionsGet##type(nullptr, nullptr, petsc_string, &variable, &set); \
185  if (!set) \
186  PetscOptionsSetValue(nullptr, petsc_string, std::to_string(variable).c_str());
187 
188 #define PETSC_OPTION_STRING(petsc_string, variable) \
189  variable.resize(MAX_STRING_LENGTH); \
190  PetscOptionsGetString(nullptr, nullptr, petsc_string, variable.data(), MAX_STRING_LENGTH, &set); \
191  if (!set) \
192  PetscOptionsSetValue(nullptr, petsc_string, variable.c_str()); \
193  variable.resize(std::strlen(variable.data()));
194  // we do variable.resize(MAX_STRING_LENGTH) to make sure there is enough memory to store the option name.
195  // may otherwise result in a segfault
196  // after this, we resize to the right size to avoid bugs with info()
197 
198  Runner runner = this->runner();
199 
200  if (pm->isParallel()) {
201  m_mat_type = "mpiaij";
202  m_vec_type = "mpi";
203  }
204  else {
205  m_mat_type = "seqaij";
206  m_vec_type = "seq";
207  }
208 
209  if (runner.isInitialized()) {
210  switch (runner.executionPolicy()) {
211  case Accelerator::eExecutionPolicy::CUDA:
212  m_vec_type += "cuda";
213  m_mat_type += "cusparse";
214  break;
215  case Accelerator::eExecutionPolicy::HIP:
216  m_vec_type += "hip";
217  m_mat_type += "hipsparse";
218  break;
219  default:
220  break;
221  }
222  }
223 
224  PetscBool set;
225  PETSC_OPTION(Real, "-ksp_rtol", m_ksp_rtol)
226  PETSC_OPTION(Real, "-ksp_atol", m_ksp_atol)
227  PETSC_OPTION(Int, "-ksp_max_it", m_ksp_max_it)
228  PETSC_OPTION_STRING("-ksp_type", m_ksp_type)
229  PETSC_OPTION_STRING("-pc_type", m_pc_type)
230  PETSC_OPTION_STRING("-mat_type", m_mat_type)
231  PETSC_OPTION_STRING("-vec_type", m_vec_type)
232 
233  info() << "[PETSc-Info] Using " << std::string(m_mat_type.c_str()) << " matrix type";
234  info() << "[PETSc-Info] Using " << std::string(m_vec_type.c_str()) << " vector type";
235 }

244 void PETScDoFLinearSystemImpl::
245 _computeMatrixNumeration()
246 {
247  // TODO use ISLocalToGlobalMappingCreate PETSc struct
248  IItemFamily* dof_family = dofFamily();
249  IParallelMng* pm = dof_family->parallelMng();
250  const bool is_parallel = pm->isParallel();
251  const Int32 nb_rank = pm->commSize();
252  const Int32 my_rank = pm->commRank();
253 
254  DoFGroup all_dofs = dof_family->allItems();
255  DoFGroup own_dofs = all_dofs.own();
256  m_nb_own_row = own_dofs.size();
257  m_nb_total_row = all_dofs.size();
258  m_first_row = 0;
259 
260  if (is_parallel) {
261  // TODO: utiliser un Scan lorsque ce sera disponible dans Arcane
262  UniqueArray<Int32> parallel_rows_index(nb_rank, 0);
263  pm->allGather(ConstArrayView<Int32>(1, &m_nb_own_row), parallel_rows_index);
264  // info() << "ALL_NB_ROW = " << parallel_rows_index;
265  m_nb_total_row = 0;
266  // TODO optimize partial and total sum
267  for (Int32 v : parallel_rows_index)
268  m_nb_total_row += v;
269  for (Int32 i = 0; i < my_rank; ++i)
270  m_first_row += parallel_rows_index[i];
271  }
272 
273  // TODO api accelerator
274  ENUMERATE_DOF (idof, own_dofs) {
275  DoF dof = *idof;
276  m_dof_matrix_numbering[idof] = m_first_row + idof.index();
277  // info() << "Numbering dof_uid=" << dof.uniqueId() << " M=" << m_dof_matrix_numbering[idof];
278  }
279 
280  m_dof_matrix_numbering.synchronize();
281  pm->barrier();
282 
283  m_parallel_rows_index.resize(m_nb_own_row);
284  m_result_work_values.resize(m_nb_own_row);
285  m_rhs_work_values.resize(m_nb_own_row);
286 }

297 void PETScDoFLinearSystemImpl::
298 _preallocateMatrix()
299 {
300  IItemFamily* dof_family = dofFamily();
301  IParallelMng* pm = dof_family->parallelMng();
302  Runner runner = this->runner();
303  MPI_Comm mpi_comm = static_cast<MPI_Comm>(pm->communicator());
304  DoFGroup all_dofs = dof_family->allItems();
305  PetscInt local_rows = m_nb_own_row; // rows this rank owns
306  PetscInt global_rows = m_nb_total_row; // total rows across all ranks
307  CSRFormatView csr_view = this->getCSRValues();
308 
309  NumArray<PetscInt, MDDim1> indices{ all_dofs.size() };
310  // info() << "nb total row " << m_nb_total_row;
311 
312  ENUMERATE_DOF (idof, all_dofs) {
313  indices[idof.index()] = m_dof_matrix_numbering[idof];
314  // info() << "local index: " << idof.index() << " global index: " << indices[idof.index()];
315  }
316 
317  PetscCallAbort(mpi_comm, ISLocalToGlobalMappingCreate(mpi_comm, 1, all_dofs.size(), indices._internalData(), PETSC_COPY_VALUES, &m_petsc_map));
318 
319  // info() << "Total " << m_nb_total_row << " local: " << m_nb_own_row;
320  pm->barrier();
321 
322  PetscCallAbort(mpi_comm, MatCreate(mpi_comm, &m_petsc_matrix));
323  PetscCallAbort(mpi_comm, MatSetSizes(m_petsc_matrix, local_rows, local_rows, global_rows, global_rows));
324  PetscCallAbort(mpi_comm, MatSetFromOptions(m_petsc_matrix));
325  PetscCallAbort(mpi_comm, MatSetLocalToGlobalMapping(m_petsc_matrix, m_petsc_map, m_petsc_map));
326 
327  // info() << "nb cols: " << csr_view.nbColumn() << ", nb rows: " << csr_view.nbRow() << ", nb vals: " << csr_view.nbValue();
328 
329  // We use COO for PETSc, so we need to convert from CSR to COO
330  RunQueue queue = makeQueue(runner);
331  NumArray<PetscInt, MDDim1> coo_rows;
332  coo_rows.resize(csr_view.nbValue());
333  _translateCSRToCOO(csr_view.rows(), coo_rows.to1DSmallSpan(), queue);
334 
335  NumArray<PetscInt, MDDim1> coo_cols;
336  //NumArray<PetscInt, MDDim1> coo_cols;
337  coo_cols.resize(csr_view.columns().size());
338  MemoryUtils::copy(coo_cols.to1DSpan(), csr_view.columns(), &queue);
339 
340  //NumArray<PetscInt, MDDim1> coo_cols(csr_view.columns());
341  //coo_cols.copy(csr_view.columns(), queue); // copy column array
342 
343  PetscCallAbort(mpi_comm, MatSetPreallocationCOOLocal(m_petsc_matrix, csr_view.nbValue(), coo_rows.to1DSpan().data(), coo_cols.to1DSpan().data()));
344  PetscCallAbort(mpi_comm, MatAssemblyBegin(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
345  PetscCallAbort(mpi_comm, MatAssemblyEnd(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
346 
347  PetscCallAbort(mpi_comm, KSPCreate(mpi_comm, &m_petsc_solver_context));
348  PetscCallAbort(mpi_comm, KSPSetOperators(m_petsc_solver_context, m_petsc_matrix, m_petsc_matrix));
349  PetscCallAbort(mpi_comm, KSPSetFromOptions(m_petsc_solver_context));
350 }

378 void PETScDoFLinearSystemImpl::
379 _initSolve()
380 {
381  IItemFamily* dof_family = dofFamily();
382  IParallelMng* pm = dof_family->parallelMng();
383  Runner runner = this->runner();
384  MPI_Comm mpi_comm = static_cast<MPI_Comm>(pm->communicator());
385  CSRFormatView csr_view = this->getCSRValues();
386 
387  if (isMatrixValuesConstant() && !isMatrixSparsityConstant())
388  // PetscCallAbort(mpi_comm, PetscError(mpi_comm, __LINE__, "_initSolve", __FILE__, PETSC_ERR_SUP, PETSC_ERROR_INITIAL, "Cannot have constant matrix values and variable matrix sparsity."));
389  ARCANE_THROW(NotSupportedException, "Cannot have constant matrix values and variable matrix sparsity.");
390 
391  _handleParameters(pm);
392  _computeMatrixNumeration();
393 
394  if (isMatrixSparsityConstant())
395  _preallocateMatrix();
396 
397  if (isMatrixValuesConstant()) {
398  PetscCallAbort(mpi_comm, MatSetValuesCOO(m_petsc_matrix, csr_view.values().data(), INSERT_VALUES));
399  PetscCallAbort(mpi_comm, MatAssemblyBegin(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
400  PetscCallAbort(mpi_comm, MatAssemblyEnd(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
401  }
402 
403  m_is_initialized = true;
404 }

427 void PETScDoFLinearSystemImpl::
428 solve()
429 {
430  info() << "[PETSc-Info] Calling PETSc solver";
431 
432  if (!m_is_initialized)
433  _initSolve();
434 
435  IItemFamily* dof_family = dofFamily();
436  IParallelMng* pm = dof_family->parallelMng();
437  Runner runner = this->runner();
438  MPI_Comm mpi_comm = static_cast<MPI_Comm>(pm->communicator());
439 
440  Parallel::Communicator arcane_comm = pm->communicator();
441  if (arcane_comm.isValid())
442  mpi_comm = static_cast<MPI_Comm>(arcane_comm);
443 
444  bool is_parallel = pm->isParallel();
445  CSRFormatView csr_view = this->getCSRValues();
446 
447  PetscInt local_rows = m_nb_own_row; // rows this rank owns
448  PetscInt global_rows = m_nb_total_row; // total rows across all ranks
449  bool is_use_device = false;
450 
451  if (runner.isInitialized()) {
452  is_use_device = isAcceleratorPolicy(runner.executionPolicy());
453  info() << "[PETSc-Info] Runner for PETSc=" << runner.executionPolicy() << " wanted_is_device=" << is_use_device;
454  }
455 
456  Real c1 = platform::getRealTime();
457 
458  // TODO: use COO with MatSetPreallocationCOO for better performance
459  // TODO: see if we pass pointers to device memory directly when is_use_device==true
460  if (!isMatrixSparsityConstant())
461  _preallocateMatrix();
462 
463  if (!isMatrixValuesConstant()) {
464  PetscCallAbort(mpi_comm, MatSetValuesCOO(m_petsc_matrix, csr_view.values().data(), INSERT_VALUES));
465  PetscCallAbort(mpi_comm, MatAssemblyBegin(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
466  PetscCallAbort(mpi_comm, MatAssemblyEnd(m_petsc_matrix, MAT_FINAL_ASSEMBLY));
467  }
468 
469  pm->barrier();
470 
471  Real b2 = platform::getRealTime();
472 
473  info() << "[PETSc-Timer] Time to create matrix = " << (b2 - c1);
474 
475  VariableDoFReal& rhs_variable = this->rhsVariable();
476  VariableDoFReal& dof_variable = this->solutionVariable();
477  const Real* rhs_data = rhs_variable.asArray().data();
478  const Real* result_data = dof_variable.asArray().data();
479 
480  Span<const Int32> rows_index_span = m_dof_matrix_numbering.asArray();
481  const Int32* rows_index_data = rows_index_span.data();
482 
483  Real b1 = platform::getRealTime();
484 
485  if (is_parallel) {
486 #if PETSC_VERSION_GE(3, 20, 0)
487  PetscCallAbort(mpi_comm, VecCreateFromOptions(mpi_comm, nullptr, 1, local_rows, global_rows, &m_petsc_rhs_vector));
488  PetscCallAbort(mpi_comm, VecCreateFromOptions(mpi_comm, nullptr, 1, local_rows, global_rows, &m_petsc_solution_vector));
489 #else
490  PetscCallAbort(mpi_comm, VecCreateMPI(mpi_comm, local_rows, global_rows, &m_petsc_rhs_vector));
491  PetscCallAbort(mpi_comm, VecSetFromOptions(m_petsc_rhs_vector));
492  PetscCallAbort(mpi_comm, VecCreateMPI(mpi_comm, local_rows, global_rows, &m_petsc_solution_vector));
493  PetscCallAbort(mpi_comm, VecSetFromOptions(m_petsc_solution_vector));
494 #endif
495  PetscCallAbort(mpi_comm, VecSetValues(m_petsc_rhs_vector, dof_variable.asArray().size(), rows_index_data, rhs_data, ADD_VALUES));
496  PetscCallAbort(mpi_comm, VecSetValues(m_petsc_solution_vector, dof_variable.asArray().size(), rows_index_data, result_data, ADD_VALUES));
497  }
498  else {
499  PetscCallAbort(mpi_comm, VecCreateSeqWithArray(mpi_comm, 1, global_rows, rhs_data, &m_petsc_rhs_vector));
500  PetscCallAbort(mpi_comm, VecCreateSeqWithArray(mpi_comm, 1, global_rows, result_data, &m_petsc_solution_vector));
501  PetscCallAbort(mpi_comm, VecSetFromOptions(m_petsc_rhs_vector));
502  PetscCallAbort(mpi_comm, VecSetFromOptions(m_petsc_solution_vector));
503  }
504 
505  pm->barrier();
506 
507  PetscCallAbort(mpi_comm, VecAssemblyBegin(m_petsc_rhs_vector));
508  PetscCallAbort(mpi_comm, VecAssemblyEnd(m_petsc_rhs_vector));
509  PetscCallAbort(mpi_comm, VecAssemblyBegin(m_petsc_solution_vector));
510  PetscCallAbort(mpi_comm, VecAssemblyEnd(m_petsc_solution_vector));
511 
512  Real a1 = platform::getRealTime();
513  info() << "[PETSc-Timer] Time to create vectors = " << (a1 - b1);
514 
515  PetscCallAbort(mpi_comm, KSPSolve(m_petsc_solver_context, m_petsc_rhs_vector, m_petsc_solution_vector));
516  Real a2 = platform::getRealTime();
517  info() << "[PETSc-Timer] Time to solve = " << (a2 - a1);
518 
519  PetscInt iteration_idx;
520  PetscCallAbort(mpi_comm, KSPGetIterationNumber(m_petsc_solver_context, &iteration_idx));
521 
522  info() << "[PETSc-Info] Used " << m_pc_type << " preconditionner. Converged in " << iteration_idx + 1 << " iterations";
523 
524  if (is_parallel) {
525  // Fill 'm_parallel_rows_index' with only rows we owns
526  // NOTE: This is only needed if matrix structure has changed.
527  Int32 index = 0;
528  ENUMERATE_ (DoF, idof, dof_family->allItems()) {
529  DoF dof = *idof;
530  if (!dof.isOwn())
531  continue;
532  m_parallel_rows_index[index] = rows_index_span[idof.index()];
533  ++index;
534  }
535  }
536 
537  if (is_parallel) {
538  Int32 nb_wanted_row = m_parallel_rows_index.extent0();
539 
540  PetscCallAbort(mpi_comm, VecGetValues(m_petsc_solution_vector, nb_wanted_row, m_parallel_rows_index.to1DSpan().data(), m_result_work_values.to1DSpan().data()));
541  // for (int i = 0; i < nb_wanted_row; i++)
542  // info() << "rows: " << m_parallel_rows_index[i];
543 
544  ENUMERATE_ (DoF, idof, dof_family->allItems().own()) {
545  // Int32 global_idx = rows_index_span[idof.index()];
546  // info() << "u[" << global_idx << "] = " << m_result_work_values[idof.index()];
547  dof_variable[idof] = m_result_work_values[idof.index()];
548  }
549  }
550  else {
551  const PetscScalar* vals = nullptr;
552  PetscCallAbort(mpi_comm, VecGetArrayRead(m_petsc_solution_vector, &vals));
553 
554  // Copy directly into Arcane DoF variable
555  dof_variable.asArray().copy(Span<const Real>(vals, m_nb_own_row));
556  PetscCallAbort(mpi_comm, VecRestoreArrayRead(m_petsc_solution_vector, &vals));
557  }
558 
559  auto a = runner.deviceMemoryInfo();
560 
561  info() << "[PETSc-Info] Wrote solution in solution_variable";
562  info() << "[PETSc-Info] Device memory allocation (Mo): " << (a.totalMemory() - a.freeMemory()) / 1e6;
563 
564  if (!isMatrixSparsityConstant()) {
565  PetscCallAbort(mpi_comm, ISLocalToGlobalMappingDestroy(&m_petsc_map));
566  PetscCallAbort(mpi_comm, MatDestroy(&m_petsc_matrix));
567  PetscCallAbort(mpi_comm, KSPDestroy(&m_petsc_solver_context));
568  }
569 
570  PetscCallAbort(mpi_comm, VecDestroy(&m_petsc_solution_vector));
571  PetscCallAbort(mpi_comm, VecDestroy(&m_petsc_rhs_vector));
572 }
