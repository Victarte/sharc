614 void FemModuleElasticity::
615 _computePrePro()
616 {
617  const IGeometryMng::RealVariable & cell_measure =
618  m_geometry_mng->getRealVariableProperty(mesh()->allCells(),IGeometryProperty::PMeasure);
619 
620  // Compute BodyForce : Grad(Pressure)
621  if (mesh()->dimension() == 3)
622  {
623  if (m_hex_quad_mesh)
624  {
625  ENUMERATE_NODE(inode,allNodes())
626  {
627  Real pressure = 0. ;
628  Real volume = 0. ;
629  ENUMERATE_CELL(icell,inode->cells())
630  {
631  Real cell_vol = cell_measure[icell]/icell->nbNode() ;
632  volume += cell_vol ;
633  pressure += cell_vol*m_cell_pressure[icell] ;
634  }
635  m_node_pressure[inode] = pressure/volume ;
636  }
637  ENUMERATE_CELL(icell,allCells())
638  {
639  m_cell_body_force[icell] = - ArcaneFemFunctions::FeOperation3D::computeGradientHexa8(*icell,m_node_coord,m_node_pressure) ;
640  }
641  }

899 void FemModuleElasticity::
900 _assembleBilinearOperatorCpu(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
901 {
902  const Int32 dim = mesh()->dimension();
903  auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
904 
905  ENUMERATE_ (Cell, icell, allCells())
906  {
907  Cell cell = *icell;
908  auto K_e = compute_element_matrix(cell);
909 
910  Int32 n1_index = 0;
911  for (Node node1 : cell.nodes())
912  {
913  if (node1.isOwn())
914  {
915  Int32 n2_index = 0;
916  for (Node node2 : cell.nodes()) {
917  for (Int32 i = 0; i < dim; ++i) {
918  DoFLocalId dof1 = node_dof.dofId(node1, i);
919  for (Int32 j = 0; j < dim; ++j) {
920  DoFLocalId dof2 = node_dof.dofId(node2, j);
921  Real value = K_e(dim * n1_index + i, dim * n2_index + j);
922  m_linear_system.matrixAddValue(dof1, dof2, value);
923  }
924  }
925  ++n2_index;
926  }
927  }
928  ++n1_index;
929  }
930  }
931 }
