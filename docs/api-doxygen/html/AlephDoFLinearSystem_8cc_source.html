262 void AlephDoFLinearSystemImpl::
263 _fillRowColumnEliminationInfos()
264 {
265  OrderedRowColumnMap& rc_elimination_map = _rowColumnEliminationMap();
266  rc_elimination_map.clear();
267  DoFInfoListView item_list_view(dofFamily());
268 
269  auto& dof_elimination_info = getEliminationInfo();
270  auto& dof_elimination_value = getEliminationValue();
271 
272  for (const auto& rc_value : m_values_map) {
273  RowColumn rc = rc_value.first;
274  Real value = rc_value.second;
275  DoF dof_row = item_list_view[rc.row_id];
276  DoF dof_column = item_list_view[rc.column_id];
277  Byte row_elimination_info = dof_elimination_info[dof_row];
278  Byte column_elimination_info = dof_elimination_info[dof_column];
279  if (row_elimination_info == ELIMINATE_ROW_COLUMN || column_elimination_info == ELIMINATE_ROW_COLUMN)
280  rc_elimination_map[{ rc.row_id, rc.column_id }] = value;
281  }

287 void AlephDoFLinearSystemImpl::
288 _applyMatrixTransformationAndFillAlephMatrix()
289 {
290  _fillRowColumnEliminationInfos();
291  OrderedRowColumnMap& rc_elimination_map = _rowColumnEliminationMap();
292 
293  IItemFamily* dof_family = dofFamily();
294  DoFInfoListView item_list_view(dof_family);
295 
296  auto& dof_elimination_info = getEliminationInfo();
297  auto& dof_elimination_value = getEliminationValue();
298 
299  // Fill the matrix from the values of \a m_values_map
300  // Skip (row,column) values which are part of an elimination.
301  for (const auto& rc_value : m_values_map) {
302  RowColumn rc = rc_value.first;
303  Real value = rc_value.second;
304  DoF dof_row = item_list_view[rc.row_id];
305  DoF dof_column = item_list_view[rc.column_id];
306 
307  Byte row_elimination_info = dof_elimination_info[dof_row];
308 
309  if (row_elimination_info == ELIMINATE_ROW)
310  // Will be computed after this loop
311  continue;
312  if (rc_elimination_map.contains(rc))
313  continue;
314 
315  // Check if value is forced for current RowColumn
316  auto x = m_forced_set_values_map.find(rc);
317  if (x != m_forced_set_values_map.end()) {
318  info(4) << "FORCED VALUE R=" << rc.row_id << " C=" << rc.column_id
319  << " old=" << value << " new=" << x->second;
320  value = x->second;
321  }
322 
323  _setMatrixValue(dof_row, dof_column, value);
324  }
325 
326  const bool do_print_filling = m_do_print_filling;
327 
328  // Apply Row or Row+Column elimination on Matrix
329  // Phase 2: set the diagonal value for elimination row to 1.0
330  ENUMERATE_ (DoF, idof, dof_family->allItems()) {
331  DoF dof = *idof;
332  if (!dof.isOwn())
333  continue;
334  Byte elimination_info = dof_elimination_info[dof];
335  if (elimination_info == ELIMINATE_ROW || elimination_info == ELIMINATE_ROW_COLUMN) {
336  Real elimination_value = dof_elimination_value[dof];
337  if (do_print_filling)
338  info() << "EliminateMatrix info=" << static_cast<int>(elimination_info) << " row="
339  << std::setw(4) << dof.localId() << " value=" << elimination_value;
340  _setMatrixValue(dof, dof, 1.0);
341  }
342  }
343 }

347 void AlephDoFLinearSystemImpl::
348 _applyRHSTransformation()
349 {
350  const bool do_print_filling = m_do_print_filling;
351 
352  // Apply Row+Column elimination
353  // Phase 1:
354  // - subtract values of the RHS vector if Row+Column elimination
355  _applyRowColumnEliminationToRHS(do_print_filling);
356 
357  IItemFamily* dof_family = dofFamily();
358 
359  auto& dof_elimination_info = getEliminationInfo();
360  auto& dof_elimination_value = getEliminationValue();
361  auto& rhs_variable = rhsVariable();
362 
363  // Apply Row or Row+Column elimination on RHS
364  ENUMERATE_ (DoF, idof, dof_family->allItems()) {
365  DoF dof = *idof;
366  if (!dof.isOwn())
367  continue;
368  Byte elimination_info = dof_elimination_info[dof];
369  if (elimination_info == ELIMINATE_ROW || elimination_info == ELIMINATE_ROW_COLUMN) {
370  Real elimination_value = dof_elimination_value[dof];
371  rhs_variable[dof] = elimination_value;
372  if (do_print_filling)
373  info() << "EliminateRHS info=" << static_cast<int>(elimination_info) << " row="
374  << std::setw(4) << dof.localId() << " value=" << elimination_value;
375  }
376  }

397 void AlephDoFLinearSystemImpl::
398 _fillRHSVector()
399 {
400  ARCANE_CHECK_POINTER(m_aleph_rhs_vector);
401 
402  // For the LinearSystem class we need an array
403  // with only the values for the ownNodes().
404  // The values of 'rhs_values' should not be updated after
405  // this call.
406  UniqueArray<Real> rhs_values_for_linear_system;
407  VariableDoFReal& rhs_values(rhsVariable());
408  IItemFamily* dof_family = dofFamily();
409  ENUMERATE_ (DoF, idof, dof_family->allItems().own()) {
410  Real v = rhs_values[idof];
411  if (m_do_print_filling)
412  info() << "SET VECTOR VALUE (" << std::setw(4) << idof.itemLocalId() << ") = " << v;
413  rhs_values_for_linear_system.add(rhs_values[idof]);
414  }
415 
416  m_aleph_rhs_vector->setLocalComponents(rhs_values_for_linear_system.view());
417  m_aleph_rhs_vector->assemble();
418 }

464 void AlephDoFLinearSystemImpl::
465 _computeMatrixInfo()
466 {
467  int solver_backend = static_cast<int>(m_solver_backend);
468  info() << "[AlephFem] COMPUTE_MATRIX_INFO solver_backend=" << solver_backend;
469  IParallelMng* pm = m_sub_domain->parallelMng();
470  // Aleph solver:
471  // Hypre = 2
472  // Trilinos = 3
473  // Cuda = 4 (not available)
474  // PETSc = 5
475  // We need to compile Arcane with the needed library and link
476  // the code with the associated aleph library (see CMakeLists.txt)
477  // TODO: Linear algebra backend should be accessed from arc file.
478  if (!m_aleph_kernel) {
479  info() << "Creating Aleph Kernel";
480  // We can use less than the number of MPI ranks
481  // but for the moment we use all the available cores.
482  Int32 nb_core = pm->commSize();
483  m_aleph_kernel = new AlephKernel(m_sub_domain, solver_backend, nb_core);
484  }
485  else {
486  //
487  m_need_destroy_matrix_and_vector = false;
488  }
489  IItemFamily* dof_family = dofFamily();
490  VariableDoFReal& solution_variable(solutionVariable());
491  DoFGroup own_dofs = dof_family->allItems().own();
492  m_dof_matrix_indexes.fill(-1);
493  AlephIndexing* indexing = m_aleph_kernel->indexing();
494  ENUMERATE_ (DoF, idof, own_dofs) {
495  DoF dof = *idof;
496  Integer row = indexing->get(solution_variable, dof);
497  m_dof_matrix_indexes[dof] = row;
498  }
499 
500  // Do not print information about setting matrix if matrix is too big
501  if (own_dofs.size() > 200)
502  m_do_print_filling = false;
503 }

536 void AlephDoFLinearSystemImpl::
537 solve()
538 {
539  // The matrix always need to be created if this is not explicitly done
540  if (!m_aleph_matrix)
541  applyMatrixTransformation();
542 
543  _createRHSAndSolutionVector();
544  _fillRHSVector();
545 
546  info() << "Calling AlephDoFLinearSystemImpl::solve()";
547  UniqueArray<Real> aleph_result;
548 
549  IItemFamily* dof_family = dofFamily();
550  DoFGroup own_dofs = dof_family->allItems().own();
551  const Int32 nb_dof = own_dofs.size();
552  m_vector_zero.resize(nb_dof);
553  m_vector_zero.fill(0.0);
554 
555  m_aleph_solution_vector->setLocalComponents(m_vector_zero);
556  m_aleph_solution_vector->assemble();
557 
558  Int32 nb_iteration = 0;
559  Real residual_norm = 0.0;
560  info() << "[AlephFem] BEGIN SOLVING WITH ALEPH solver_backend=" << static_cast<int>(m_solver_backend);
561 
562  // Post the solver. The call is asynchronous, and we wait for the result
563  // when calling syncSolver().
564  // The values nb_iteration and residual_norm are not used in this case.
565  // We get them during the call to syncSolver()
566  m_aleph_matrix->solve(m_aleph_solution_vector, m_aleph_rhs_vector,
567  nb_iteration, &residual_norm,
568  m_aleph_params, true);
569 
570  // Reset matrix and vectors because there can no longer be used
571  // They will be re-created when needed if we call solve() again.
572  // NOTE: it is the aleph library we do not need to call delete() because
573  m_aleph_rhs_vector = nullptr;
574  m_aleph_solution_vector = nullptr;
575  m_aleph_matrix = nullptr;
576 
577  info() << "[AlephFem] END SOLVING WITH ALEPH r=" << residual_norm
578  << " nb_iter=" << nb_iteration;
579 
580  // Wait for the solver to finish and get solution vector
581  auto* solution_vector = m_aleph_kernel->syncSolver(0, nb_iteration, &residual_norm);
582 
583  solution_vector->getLocalComponents(aleph_result);
584 
585  const bool do_verbose = (nb_dof < 200);
586  Int32 index = 0;
587 
588  VariableDoFReal& solution_variable(this->solutionVariable());
589  ENUMERATE_ (DoF, idof, dofFamily()->allItems().own()) {
590  DoF dof = *idof;
591 
592  solution_variable[dof] = aleph_result[m_aleph_kernel->indexing()->get(solution_variable, dof)];
593  if (do_verbose)
594  info() << "Node uid=" << dof.uniqueId() << " V=" << aleph_result[index] << " T=" << solution_variable[dof];
595  ++index;
596  }
597 }
