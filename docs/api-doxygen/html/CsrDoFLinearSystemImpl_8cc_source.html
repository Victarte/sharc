51 _applyForcedValuesToLhs()
52 {
53  IItemFamily* dof_family = dofFamily();
54  auto nb_dof = dof_family->nbItem();
55 
56  RunQueue queue = makeQueue(runner());
57  auto command = makeCommand(queue);
58 
59  auto in_out_forced_info = Accelerator::viewInOut(command, getForcedInfo());
60  auto in_out_forced_value = Accelerator::viewInOut(command, getForcedValue());
61  auto csr_view = m_csr_view;
62 
63  command << RUNCOMMAND_LOOP1(iter, nb_dof)
64  {
65  auto [i] = iter();
66  DoFLocalId dof_id(i);
67  if (in_out_forced_info[dof_id]) {
68  auto index = csr_view.tryFindColumnInRow(dof_id, dof_id);
69  csr_view.value(index) = in_out_forced_value[dof_id];
70  }
71  };

88 void CsrDoFLinearSystemImpl::
89 _applyRowColumnEliminationOnMatrix()
90 {
91  IItemFamily* dof_family = dofFamily();
92 
93  auto nb_dof = dof_family->nbItem();
94 
95  RunQueue queue = makeQueue(runner());
96  auto command = makeCommand(queue);
97 
98  auto in_elimination_info = Accelerator::viewIn(command, getEliminationInfo());
99  auto in_elimination_value = Accelerator::viewIn(command, getEliminationValue());
100 
101  auto in_out_rhs_variable = Accelerator::viewInOut(command, rhsVariable());
102  auto csr_view = m_csr_view;
103  command << RUNCOMMAND_LOOP1(iter, nb_dof)
104  {
105  auto [row_index] = iter();
106  DoFLocalId dof_row(row_index);
107  auto row_elimination_info = in_elimination_info[dof_row];
108  bool is_row_elimination = (row_elimination_info == ELIMINATE_ROW) || (row_elimination_info == ELIMINATE_ROW_COLUMN);
109  for (CsrRowColumnIndex csr_index : csr_view.rowRange(dof_row)) {
110  Int32 column_index = csr_view.column(csr_index);
111  if (column_index > 0) {
112  DoFLocalId dof_column(column_index);
113  auto column_elimination_info = in_elimination_info[dof_column];
114  bool is_column_elimination = (column_elimination_info == ELIMINATE_ROW) || (column_elimination_info == ELIMINATE_ROW_COLUMN);
115  if (is_row_elimination || is_column_elimination) {
116  csr_view.value(csr_index) = (dof_column == dof_row) ? 1.0 : 0.0;
117  }
118  }
119  }
120  };
121 }

126 void CsrDoFLinearSystemImpl::
127 _applyRowEliminationOnMatrix()
128 {
129  IItemFamily* dof_family = dofFamily();
130 
131  auto nb_dof = dof_family->nbItem();
132 
133  RunQueue queue = makeQueue(runner());
134  auto command = makeCommand(queue);
135 
136  auto in_elimination_info = Accelerator::viewIn(command, getEliminationInfo());
137  auto in_elimination_value = Accelerator::viewIn(command, getEliminationValue());
138 
139  auto in_out_rhs_variable = Accelerator::viewInOut(command, rhsVariable());
140  auto csr_view = m_csr_view;
141  command << RUNCOMMAND_LOOP1(iter, nb_dof)
142  {
143  auto [thread_id] = iter();
144  DoFLocalId dof_id(thread_id);
145  auto elimination_info = in_elimination_info[dof_id];
146  if (elimination_info == ELIMINATE_ROW) {
147  auto elimination_value = in_elimination_value[dof_id];
148  for (CsrRowColumnIndex csr_index : csr_view.rowRange(dof_id))
149  csr_view.value(csr_index) = (csr_view.column(csr_index) == dof_id) ? 1.0 : 0.0;
150  }
151  };
152 }

187 void CsrDoFLinearSystemImpl::
188 _fillRowColumnEliminationInfos()
189 {
190  // TODO: add a way to detect if user has added row/column elimination
191  // to avoid doing this loop if this is not needed (because at the moment it
192  // is only on CPU because OrderedRowColumnMap need to be ordered and so is
193  // not available on GPU)
194  OrderedRowColumnMap& rc_elimination_map = _rowColumnEliminationMap();
195  rc_elimination_map.clear();
196  DoFInfoListView item_list_view(dofFamily());
197 
198  auto& dof_elimination_info = getEliminationInfo();
199  auto& dof_elimination_value = getEliminationValue();
200  auto csr_view = m_csr_view;
201  const Int32 nb_dof = dofFamily()->nbItem();
202 
203  m_has_row_column_elimination = false;
204  for (Int32 i = 0; i < nb_dof; ++i) {
205  DoFLocalId dof_row(i);
206  Byte row_elimination_info = dof_elimination_info[dof_row];
207  if (row_elimination_info == ELIMINATE_ROW_COLUMN) {
208  m_has_row_column_elimination = true;
209  break;
210  }
211  }
212  if (!m_has_row_column_elimination)
213  return;
214 
215  for (Int32 i = 0; i < nb_dof; ++i) {
216  DoFLocalId dof_row(i);
217  for (CsrRowColumnIndex csr_index : csr_view.rowRange(dof_row)) {
218  Int32 col_index = csr_view.column(csr_index);
219  if (col_index >= 0) {
220  DoFLocalId dof_column(col_index);
221  Byte row_elimination_info = dof_elimination_info[dof_row];
222  Byte column_elimination_info = dof_elimination_info[dof_column];
223  if (row_elimination_info == ELIMINATE_ROW_COLUMN || column_elimination_info == ELIMINATE_ROW_COLUMN) {
224  Real value = csr_view.value(csr_index);
225  rc_elimination_map[{ i, col_index }] = value;
226  }
227  }
228  }
229  }
230 }
