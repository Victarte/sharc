103 void GaussDoFsOnCells::Impl::
104 initialize(IMesh* mesh, Int32 max_nb_gauss_per_cell)
105 {
106  IItemFamily* dof_family_interface = mesh->findItemFamily(Arcane::IK_DoF, "GaussCellFamily", true);
107  mesh::DoFFamily* dof_family = ARCANE_CHECK_POINTER(dynamic_cast<mesh::DoFFamily*>(dof_family_interface));
108  m_gauss_family = dof_family_interface;
109 
110  // Create the Gauss points as Arcane "DoFs" attached to cells
111  Int64UniqueArray uids(mesh->allCells().size() * max_nb_gauss_per_cell);
112  Int64 max_cell_uid = mesh::DoFUids::getMaxItemUid(mesh->cellFamily());
113  {
114  Integer gauss_index{ 0 };
115  ENUMERATE_CELL (icell, mesh->allCells()) {
116  Cell cell = *icell;
117  auto cell_type = cell.type();
118  Int64 cell_unique_id = cell.uniqueId().asInt64();
119  for (Integer i = 0; i < max_nb_gauss_per_cell; ++i) {
120  uids[gauss_index++] = cell_unique_id * max_nb_gauss_per_cell + i;
121  }
122  }
123  }
124 
125  Integer uidsize = uids.size();
126  Int32UniqueArray gauss_lids(uidsize);
127  dof_family->addDoFs(uids, gauss_lids);
128  dof_family->endUpdate();
129  Integer nb_gauss = dof_family->allItems().size();
130  info() << "NB_GAUSS=" << nb_gauss << " max_per_cell=" << max_nb_gauss_per_cell;
131 
132  // Create Cell -> Gauss (DoF) connectivity.
133  m_cell_gauss_connectivity = mesh->indexedConnectivityMng()->findOrCreateConnectivity(mesh->cellFamily(), dof_family, "GaussCell");
134  auto* cn = m_cell_gauss_connectivity->connectivity();
135  {
136  Integer gauss_index{ 0 };
137  ENUMERATE_CELL (icell, mesh->allCells()) {
138  CellLocalId cell = *icell;
139  for (Integer i = 0; i < max_nb_gauss_per_cell; ++i) {
140  cn->addConnectedItem(cell, DoFLocalId(gauss_lids[gauss_index++]));
141  }
142  }
143  }
144  info() << "End build Gauss points";
145 
146  IndexedCellDoFConnectivityView cell_gauss(m_cell_gauss_connectivity->view());
147  {
148  // Set the owners of the Gauss point (=DoF).
149  IParallelMng* pm = mesh->parallelMng();
150  Int32 my_rank = pm->commRank();
151  ItemInternalArrayView gauss_dofs = m_gauss_family->itemsInternal();
152  ENUMERATE_CELL (icell, mesh->allCells()) {
153  Cell cell = *icell;
154  Int32 cell_owner = cell.owner();
155  for (DoFLocalId gauss : cell_gauss.dofs(cell)) {
156  gauss_dofs[gauss]->setOwner(cell_owner, my_rank);
157  }
158  }
159  dof_family->notifyItemsOwnerChanged();
160  dof_family->computeSynchronizeInfos();
161  }
162 }
